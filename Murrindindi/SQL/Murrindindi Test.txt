select
    auprparc.pcl_num,
    auprparc.pcl_flg,
    auprparc.ass_num,
    auprparc.str_seq,
    auprstad.seq_num,
    auprstad.hou_num,
    auprstad.hou_alp,
    auprstad.hou_end,
    auprstad.end_alp,
    auprstad.str_nme,
    auprstad.str_typ,
    auprstad.sbr_nme
from
    AUTHORITY_auprparc as auprparc ,
    AUTHORITY_auprstad as auprstad
where
    auprparc.pcl_num = auprstad.pcl_num AND
    auprparc.pcl_flg in ( 'R' , 'P' ) AND
    auprparc.ass_num is not null
