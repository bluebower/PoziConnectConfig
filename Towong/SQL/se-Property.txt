SELECT [Lynx_Propertys].[Property],
  [Lynx_Propertys].[Name] AS [Property_Name],
  [Lynx_Propertys].[FloorNo],
  [Lynx_Propertys].[UnitNo],
  [Lynx_Propertys].[StreetNoFrom],
  [Lynx_Propertys].[StreetNoTo],
  [Lynx_Streets].[Name] AS [Street_Name],
  [Lynx_StreetType].[Type] AS [Street_Type],
  [Lynx_Localities].[Locality],
  [Lynx_Localities].[Postcode]
FROM [Lynx_Propertys]
  INNER JOIN [Lynx_Streets] ON [Lynx_Propertys].[StreetID] = [Lynx_Streets].[ID]
  INNER JOIN [Lynx_StreetType] ON [Lynx_Streets].[Type] = [Lynx_StreetType].[ID]
  INNER JOIN [Lynx_Localities] ON [Lynx_Streets].[Locality] =
    [Lynx_Localities].[ID]
WHERE [Lynx_Propertys].[LandClassificationCode] <> '10' AND
  [Lynx_Propertys].[Type] <> '672'
